    function setupAutocomplete(inputEl, suggestionsEl, endpoint) {
        let timer = null;

        inputEl.addEventListener('input', () => {
            const value = inputEl.value.trim();
            clearTimeout(timer);

            if (value.length < 2) {
                suggestionsEl.innerHTML = '';
                suggestionsEl.style.display = 'none';
                return;
            }

            timer = setTimeout(async () => {
                try {
                    const resp = await fetch(endpoint + '?q=' + encodeURIComponent(value), {
                        headers: getAuthHeaders(false)
                    });
                    if (!resp.ok) {
                        suggestionsEl.innerHTML = '';
                        suggestionsEl.style.display = 'none';
                        return;
                    }
                    const list = await resp.json();
                    if (!list || list.length === 0) {
                        suggestionsEl.innerHTML = '';
                        suggestionsEl.style.display = 'none';
                        return;
                    }

                    suggestionsEl.innerHTML = '';
                    list.slice(0, 8).forEach(item => {
                        const div = document.createElement('div');
                        div.className = 'suggestion-item';
                        div.textContent = item;
                        div.onclick = () => {
                            inputEl.value = item;
                            suggestionsEl.innerHTML = '';
                            suggestionsEl.style.display = 'none';
                            inputEl.focus();
                        };
                        suggestionsEl.appendChild(div);
                    });
                    suggestionsEl.style.display = 'block';
                } catch (e) {
                    console.error(e);
                    suggestionsEl.innerHTML = '';
                    suggestionsEl.style.display = 'none';
                }
            }, 200);
        });

        document.addEventListener('click', (evt) => {
            if (!suggestionsEl.contains(evt.target) && evt.target !== inputEl) {
                suggestionsEl.innerHTML = '';
                suggestionsEl.style.display = 'none';
            }
        });
    }

    async function handleNewJobSubmit() {
        newJobError.textContent = '';

        const contractor  = contractorInput.value.trim();
        const customer    = customerInput.value.trim();
        const material    = materialInput.value.trim();
        const stage       = stageSelect.value;
        const installDate = installDateInput.value;
        let notesRaw      = notesInput.value.trim();

        if (!contractor || !customer || !material) {
            newJobError.textContent = 'Contractor, Customer and Material are required.';
            return;
        }

        const locationCode = remnantLocationSelect.value;

        notesRaw = stripOnHoldText(stripRemnantInfo(notesRaw));

        let chosenOption = 'unknown';
        remnantOptionRadios.forEach(r => {
            if (r.checked) chosenOption = r.value;
        });

        let remnantText = '';
        const remSize = remnantSizeInput.value.trim();

        if (chosenOption === 'left' && remSize) {
            remnantText = `Remnant: ${remSize} left`;
            if (locationCode) {
                remnantText += ` | Location: ${locationCode}`;
            }
        } else if (chosenOption === 'none') {
            remnantText = 'No remnants were left.';
        }

        let finalNotes = notesRaw;
        if (remnantText) {
            finalNotes = finalNotes ? (finalNotes + ' | ' + remnantText) : remnantText;
        }

        if (holdReasonContainer.style.display === 'block') {
            const reason = holdReasonInput.value.trim();
            if (reason) {
                finalNotes = finalNotes
                    ? `${finalNotes} | ON HOLD: ${reason}`
                    : `ON HOLD: ${reason}`;
            } else {
                finalNotes = finalNotes
                    ? `${finalNotes} | ON HOLD`
                    : 'ON HOLD';
            }
        }

        const body = {
            contractor,
            customer,
            material,
            stage,
            installDate: installDate || null,
            notes: finalNotes
        };

        const currentUsername = localStorage.getItem('username') || '';
        if (editingJobId === null && currentUsername) {
            body.seller = currentUsername;
        }

        try {
            let resp;
            if (editingJobId === null) {
                resp = await fetch('/api/jobs', {
                    method: 'POST',
                    headers: getAuthHeaders(true),
                    body: JSON.stringify(body)
                });
            } else {
                resp = await fetch(`/api/jobs/${editingJobId}`, {
                    method: 'PUT',
                    headers: getAuthHeaders(true),
                    body: JSON.stringify(body)
                });
            }

            if (!resp.ok) {
                const text = await resp.text();
                newJobError.textContent = 'Error saving job: HTTP ' + resp.status + ' - ' + text;
                return;
            }

            resetForm();
            await loadAll();
        } catch (err) {
            newJobError.textContent = 'Network error: ' + err;
        }
    }

    async function loadRemnants() {
        remnantsError.textContent = '';
        const tbody = document.getElementById('remnantsBody');
        tbody.innerHTML = '<tr><td colspan="8" class="table-empty">Loading...</td></tr>';

        try {
            const [respActive, respCompleted] = await Promise.all([
                fetch('/api/jobs/active', { headers: getAuthHeaders(false) }),
                fetch('/api/jobs/completed', { headers: getAuthHeaders(false) })
            ]);

            if (!respActive.ok || !respCompleted.ok) {
                remnantsError.textContent = `Error loading stock material.`;
                tbody.innerHTML = '<tr><td colspan="8" class="table-empty">No stock material found</td></tr>';
                return;
            }

            const activeJobs = await respActive.json() || [];
            const completedJobs = await respCompleted.json() || [];
            let all = activeJobs.concat(completedJobs);

            all = all.filter(job => hasRemnantInfo(job.notes));
            all = all.filter(job => !hasTagInfo(job.notes));     // in stock only
            all = all.filter(job => !hasUsedUp(job.notes));      // don't show used up

            const fm = filterRemnantMaterialInput.value.trim().toLowerCase();
            const fc = filterRemnantContractorInput.value.trim().toLowerCase();

            if (fm || fc) {
                all = all.filter(job => {
                    const material = (job.material || '').toLowerCase();
                    const contractor = (job.contractor || '').toLowerCase();
                    const matchMat = !fm || material.includes(fm);
                    const matchCont = !fc || contractor.includes(fc);
                    return matchMat && matchCont;
                });
            }

            if (currentRemnantLocationFilter !== 'ALL') {
                all = all.filter(job => {
                    const code = extractLocationCode(job.notes || '');
                    if (!code) return false;

                    const loc = currentRemnantLocationFilter;

                    if (loc === 'RR1' || loc === 'RR2' || loc === 'RR3' || loc === 'RR4') return code.startsWith(loc + '-');
                    if (loc === 'RR5' || loc === 'RR6') return code === loc;
                    if (loc === 'SL1' || loc === 'SL2') return code.startsWith(loc + '-');

                    return code === loc; // includes RACK-SOLD-1 and RACK-SOLD-2 now
                });
            }

            if (!all || all.length === 0) {
                tbody.innerHTML = '<tr><td colspan="8" class="table-empty">No stock material found</td></tr>';
                return;
            }

            all.sort((a, b) => {
                const da = a.installDate || '';
                const db = b.installDate || '';
                return db.localeCompare(da);
            });

            tbody.innerHTML = '';
            for (const job of all) {
                const locCode = extractLocationCode(job.notes || '');
                const locLabel = getLocationLabel(locCode);
                const isNew = isRecentRemnant(job);

                const tr = document.createElement('tr');

                    <td>${job.id}</td> 
                    <td>${job.contractor || ''}</td> 
                    <td>${job.customer || ''}</td> 
                    <td>${job.material || ''}</td> 
                    <td>${formatDate(job.installDate)}</td> 
                    <td>${locLabel || ''}</td> 
                    <td></td> 
                    <td></td> 
                `;

                const notesTd = tr.children[6];
                if (notesTd) notesTd.innerHTML = renderNotes(job.notes || '', true, isNew);

                const actionsTd = tr.children[7];

                const tagBtn = document.createElement('button');
                tagBtn.className = 'btn-sm btn-outline-success';

                    <td>${job.contractor || ''}</td> 
                    <td>${job.customer || ''}</td> 
                    <td>${job.material || ''}</td> 
                    <td>${locLabel || ''}</td> 
                    <td>${renderNotesWithTag(job.notes || '')}</td> 
                `;
                tbody.appendChild(tr);
            }
        }catch(err){
            usedUpError.textContent = 'Network error: ' + err;
            tbody.innerHTML = '<tr><td colspan="6" class="table-empty">No used up items</td></tr>';
        }
    }

    // ==============================
    // STATUS / DELETE
    // ==============================
    async function markDone(id) {
        try {
            const resp = await fetch(`/api/jobs/${id}/done`, {
                method: 'PUT',
                headers: getAuthHeaders(false)
            });
            if (!resp.ok) {
                alert('Error marking as done: HTTP ' + resp.status);
                return;
            }
            await loadAll();
        } catch (err) {
            alert('Network error: ' + err);
        }
    }

    async function undoDone(id) {
        try {
            const resp = await fetch(`/api/jobs/${id}/undo`, {
                method: 'PUT',
                headers: getAuthHeaders(false)
            });
            if (!resp.ok) {
                alert('Error undoing: HTTP ' + resp.status);
                return;
            }
            await loadAll();
        } catch (err) {
            alert('Network error: ' + err);
        }
    }

    async function deleteJob(id) {
        if (!confirm('Delete this job?')) return;
        try {
            const resp = await fetch(`/api/jobs/${id}`, {
                method: 'DELETE',
                headers: getAuthHeaders(false)
            });
            if (!resp.ok) {
                alert('Error deleting: HTTP ' + resp.status);
                return;
            }
            await loadAll();
        } catch (err) {
            alert('Network error: ' + err);
        }
    }

        await loadActive();
        await loadCompleted();
        await loadRemnants();
        await loadSoldRemnants();
        await loadUsedUpHistory();
    }
