<script>
    // ---------- LOCAL STORAGE HELPERS ----------
    function getData() {
        return JSON.parse(localStorage.getItem("remnants")) || [];
    }

    function saveData(data) {
        localStorage.setItem("remnants", JSON.stringify(data));
    }

    // ---------- ADICIONAR REMNANT ----------
    function addRemnant() {
        const remnant = {
            material: document.getElementById("material").value.trim(),
            cor: document.getElementById("cor").value.trim(),
            espessura: document.getElementById("espessura").value.trim(),
            largura: document.getElementById("largura").value.trim(),
            altura: document.getElementById("altura").value.trim(),
            sessao: document.getElementById("sessao").value,
            onde: document.getElementById("onde").value.trim(),
            obs: document.getElementById("obs").value.trim()
        };

        if (!remnant.material || !remnant.sessao || !remnant.onde) {
            alert("Preencha pelo menos Material, Sessão e Onde.");
            return;
        }

        const data = getData();
        data.push(remnant);
        saveData(data);

        // limpa campos
        document.getElementById("material").value = "";
        document.getElementById("cor").value = "";
        document.getElementById("espessura").value = "";
        document.getElementById("largura").value = "";
        document.getElementById("altura").value = "";
        document.getElementById("sessao").value = "";
        document.getElementById("onde").value = "";
        document.getElementById("obs").value = "";

        renderTable();
    }

    // ---------- RENDERIZAR TABELA ----------
    function renderTable() {
        const tbody = document.getElementById("tableBody");
        tbody.innerHTML = "";

        const data = getData();
        const fs = document.getElementById("filterSessao").value;
        const fo = document.getElementById("filterOnde").value.trim().toLowerCase();
        const fm = document.getElementById("filterMaterial").value.trim().toLowerCase();

        const filtered = data.filter(r => {
            return (!fs || r.sessao === fs) &&
                (!fo || (r.onde || "").toLowerCase().includes(fo)) &&
                (!fm || (r.material || "").toLowerCase().includes(fm));
        });

        filtered.forEach(r => {
            const tr = document.createElement("tr");
            tr.innerHTML = `
          <td>${r.material || ""}</td> 
          <td>${r.cor || ""}</td> 
          <td>${r.espessura || ""}</td> 
          <td>${r.largura || ""}</td> 
          <td>${r.altura || ""}</td> 
          <td>${r.sessao || ""}</td> 
          <td>${r.onde || ""}</td> 
          <td>${r.obs || ""}</td> 
        `;
            tbody.appendChild(tr);
        });
    }

    // ---------- BOTÃO VOLTAR PRO PAINEL ----------
    document.getElementById("backBtn").onclick = () => {
        if (document.referrer) {
            window.location.href = document.referrer;
        } else {
            window.history.back();
        }
    };

    // ---------- INICIALIZAÇÃO ----------
    renderTable();
</script>
