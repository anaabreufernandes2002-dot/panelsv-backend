<script>
    const canvas = document.getElementById("mapCanvas");
    const ctx = canvas.getContext("2d");

    const img = new Image();
    img.src = "MapaSv.png"; // sua imagem

    let scale = 1;
    let offsetX = 0;
    let offsetY = 0;

    let isPanning = false;
    let panStartX = 0;
    let panStartY = 0;

    let drawMode = false;
    let isDrawing = false;
    let currentPath = null;
    const paths = [];

    function resizeCanvas() {
        canvas.width = window.innerWidth - 20;
        canvas.height = window.innerHeight - 20;
        draw();
    }



    function clearCanvas() {
        ctx.setTransform(1,0,0,1,0,0);
        ctx.clearRect(0,0,canvas.width,canvas.height);
    }

    function draw() {
        if (!img.complete) return;

        clearCanvas();
        ctx.setTransform(scale, 0, 0, scale, offsetX, offsetY);
        ctx.drawImage(img, 0, 0);

        ctx.lineWidth = 2 / scale;
        ctx.strokeStyle = "#ef4444";

        paths.forEach(path => {
            ctx.beginPath();
            ctx.moveTo(path[0].x, path[0].y);
            for (let i = 1; i < path.length; i++) {
                ctx.lineTo(path[i].x, path[i].y);
            }
            ctx.stroke();
        });
    }
    window.addEventListener('load', () => {
        const img = document.getElementById('mapImage');
        if (img) {
            img.style.transform = 'scale(0.47)';
            img.style.transformOrigin = 'center center';
        }
    });

    function screenToWorld(x, y) {
        return {
            x: (x - offsetX) / scale,
            y: (y - offsetY) / scale
        };
    }

    function updateZoom() {
        document.getElementById("zoomLabel").textContent =
            Math.round(scale * 100) + "%";
        draw();
    }

    function resetView() {
        scale = 0.47; // zoom inicial fixo

        const imgW = img.width * scale;
        const imgH = img.height * scale;

        offsetX = (canvas.width / 2) - (imgW / 2);
        offsetY = (canvas.height / 2) - (imgH / 2);

        ctx.setTransform(1,0,0,1,0,0);
        updateZoom();
    }




    canvas.addEventListener("mousedown", (e) => {
        const rect = canvas.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;

        if (drawMode) {
            isDrawing = true;
            currentPath = [];
            currentPath.push(screenToWorld(x, y));
            paths.push(currentPath);
        } else {
            isPanning = true;
            panStartX = x - offsetX;
            panStartY = y - offsetY;
            canvas.style.cursor = "grabbing";
        }
    });

    canvas.addEventListener("mousemove", (e) => {
        const rect = canvas.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;

        if (isPanning && !drawMode) {
            offsetX = x - panStartX;
            offsetY = y - panStartY;
            draw();
        }

        if (isDrawing && drawMode) {
            currentPath.push(screenToWorld(x, y));
            draw();
        }
    });

    canvas.addEventListener("mouseup", () => {
        isPanning = false;
        isDrawing = false;
        canvas.style.cursor = drawMode ? "crosshair" : "grab";
    });

    canvas.addEventListener("wheel", (e) => {
        e.preventDefault();
        scale *= e.deltaY < 0 ? 1.1 : 0.9;
        updateZoom();
    });

    document.getElementById("zoomInBtn").onclick = () => {
        scale *= 1.1;
        updateZoom();
    };

    document.getElementById("zoomOutBtn").onclick = () => {
        scale *= 0.9;
        updateZoom();
    };

    document.getElementById("resetBtn").onclick = resetView;

    document.getElementById("drawBtn").onclick = (e) => {
        drawMode = !drawMode;
        e.target.classList.toggle("active", drawMode);
        canvas.classList.toggle("drawing", drawMode);
        canvas.style.cursor = drawMode ? "crosshair" : "grab";
    };

    // 🔙 Voltar pra página anterior (painel)
    document.getElementById("backBtn").onclick = () => {
        if (document.referrer) {
            window.location.href = document.referrer;
        } else {
            window.history.back();
        }
    };

    img.onload = () => {
        resizeCanvas();
        resetView();
        draw();
    };



    window.onresize = resizeCanvas;
</script>
